#the string in the global variables Autoload corresponding to the global variable argument
#example: global.joyJumpVal becomes global.joy_jump_val, with joy_jump_val being set as "jump" in the global variable Autoload
#this will also allow us to not have to program separate input events for controllers and keyboards

func joystick_check_button(id, input_string): #inside global variables Autoload, setting second argument as a string instead of an int
    var check = Input.is_action_pressed(input_string)
    return check
